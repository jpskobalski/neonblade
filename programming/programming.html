// Simple typewriter effect
const text = "I'm NEONBLADE, synthwave entity & code dreamer.";
let index = 0;

function type() {
  const target = document.getElementById('typeTarget');
  if (index < text.length) {
    target.textContent += text.charAt(index);
    index++;
    setTimeout(type, 50);
  }
}
